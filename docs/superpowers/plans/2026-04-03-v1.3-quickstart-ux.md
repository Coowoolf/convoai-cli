- [ ] **Step 3: Implement gradient.ts**

Create `src/ui/gradient.ts` with three exported functions:

`gradientBox({ emoji, title, subtitle, body? })` — Renders a blue→pink gradient bordered box with:
- Top/bottom borders: `╭──...──╮` / `╰──...──╯` with per-character color interpolation from `rgb(80,150,255)` to `rgb(220,72,195)`
- Left border: blue `│`, right border: pink `│`
- Title: emoji + per-character gradient text (bold)
- Subtitle: dim text
- Optional body lines (dim text)
- Returns `string[]` (array of lines)

`gradientTitle(text)` — Renders text with per-character gradient coloring. Returns `string`.

`gradientProgress(current, total)` — Renders the progress bar:
- `current/total` gradient blocks, right side empty (clean)
- Suffix: dim `N/M`
- For final step (current === total): blue→pink→green gradient
- Returns `string`

Color interpolation helper: `lerp(a, b, t)` for RGB channels.

import { describe, it, expect } from 'vitest';
import { getStrings } from '../../src/ui/i18n.js';

describe('getStrings', () => {
  it('returns Chinese strings for "cn"', () => {
    const s = getStrings('cn');
    expect(s.step1.title).toBe('配置凭证');
    expect(s.step1.emoji).toBe('🔑');
  });

  it('returns English strings for "global"', () => {
    const s = getStrings('global');
    expect(s.step1.title).toBe('Credentials');
    expect(s.step1.emoji).toBe('🔑');
  });

  it('cn and global have the same keys', () => {
    const cn = getStrings('cn');
    const en = getStrings('global');
    expect(Object.keys(cn)).toEqual(Object.keys(en));
    for (const key of Object.keys(cn)) {
      expect(Object.keys((cn as any)[key]).sort()).toEqual(Object.keys((en as any)[key]).sort());
    }
  });

  it('all steps have title, subtitle, emoji', () => {
    for (const lang of ['cn', 'global'] as const) {
      const s = getStrings(lang);
      for (const step of [s.step1, s.step2, s.step3, s.step4, s.step5, s.step6]) {
        expect(step.title).toBeTruthy();
        expect(step.subtitle).toBeTruthy();
        expect(step.emoji).toBeTruthy();
      }
    }
  });
});
